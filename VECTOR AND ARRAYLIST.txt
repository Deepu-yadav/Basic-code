        // to Arraylist but multiple thread can not access the vector only one thread access at a time
        list.add(23);
        list.add(34);
        list.add(45);
        list.add(100);
        list.add(344);
        System.out.println("it is the form of the Arraylist "+list);

        List<Integer> list1=new LinkedList<>();
        list1.add(1);
        list1.add(100);
        list1.add(200);
        System.out.println("it is the form of the LinkedList " +list1);
List<Integer> list2=new Vector<>();

list2.add(34);
list2.add(35);
list2.add(400);
        System.out.println("it is the form of vector " + list2);
    }
